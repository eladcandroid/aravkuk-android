כְּשֶׂהַנְּשָׂמָה כּוֹאֶבֶת

כַּמָּה גָּדוֹל הוּא הַכְּאֵב כְּשֶׁהַנְּשָׁמָה כּוֹאֶבֶת. מִתְרוֹמֶמֶת הִיא אֶל-עַל וְהִיא חָשָׁה כִּי כְּנָפֶיהָ נִשְׁבְּרוּ. צוֹעֶקֶת הִיא בְּקִרְבָּהּ פְּנִימָה מֵעָצְמַת מַכְאוֹבֶיהָ, אוֹי, מִי יוֹשִׁיעַ לָהּ - "מִן הַמֵּצַר קָרָאתִי יָהּ עָנָנִי בַמֶרְחַב יָהּ".

מִתְכַּנְּסִים הֵם לִפְעָמִים בְּתוֹךְ הַלֵּב כָּל יִסּוּרֵי הָעוֹלָם, כָּל עִנּוּיֵי הַנְשָׁמוֹת וְעִנּוּיֵי הַגּוּפוֹת. הָאָדָם אֵינוֹ מוֹצֵא מָנוֹחַ לִלְבָבוֹ. חָפֵץ הוּא בְּתִקּוּנוֹ שֶׁל עוֹלָם, וְהָעוֹלָם עוֹמֵד לְעֻמָּתוֹ בְּכָל כִּעוּר וְנִוּוּל, בְּכָל עֹצֶב וְיָגוֹן קוֹדֵר שֶׁלּוֹ, וְנַפְשׁוֹ בּוֹכָה בַּמִּסְתָּרִים. מִי יָנוּד לוֹ? מִי יְנַחֲמוֹ? הַמְּצִיאוּת קָשָׁה מִבַּרְזֶל וְהִיא מְטַפַּחַת עַל פָּנָיו. מִתְכַּנֵּס הָאָדָם בְּתוֹכִיּוּתוֹ כְּחֹמֶט בִּקְלִפָּתוֹ - וּמָנוֹחַ אַיִן. הוֹמֶה אָז מִקֶּרֶב נִשְׁמָתוֹ לִמְקוֹר הַיֶּשַׁע, לְמַעְיָן הָאֹשֶׁר וְהַהַרְמוֹנְיָה שֶׁל כָּל הַיֵּשׁ. "אֶל אֱלֹהִים דָּלְפָה נַפְשִׁי". כְּמַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה, בָּאָה שְׁמוּעָה טוֹבָה מִמְּקוֹר חֵי-הָעוֹלָמִים.

הֲיִתָּכֵן שֶׁלֹּא אֶמְצָא אֶת הַמָּקוֹר שֶׁל מַכְאוֹב לִבִּי, הַמִּצְטַמֵּק מִפְּנֵי אִסּוּרוֹ בְּכַבְלֵי מַחֲשַׁכִּים, בְּמַאֲסָרִים שֶׁל גְּבוּלִים וְצִמְצוּמִים? הֲיִתָּכֵן שֶׁלֹּא יָבֹא עֶזְרִי מִקֹּדֶשׁ, אַחֲרֵי כָּל יִסּוּרֵי נְשָׁמָה שֶׁאֵין לָהֶם קִצְבָּה, אַחֲרֵי תְּשׁוּקָה בּוֹעֶרֶת, אֵשׁ שֶׁאֵינָה כָּבָה, הַמְלַהֶטֶת אֶת כָּל נִימֵי נַפְשִׁי בִּפְנִימִיּוּתָהּ? וּמָה אֵלֶּה הֵמָּה מַכְאוֹבַי? רַק צִמָּאוֹן לוֹהֵט לְאוֹר ד', לְאוֹר אֱמֶת, לְאוֹר צַחְצְחוֹת עֻזְּךָ וְתִפְאַרְתֶּךָ, יוֹצְרִי וּבוֹרְאִי, רַב מְחוֹלֵל כֹּל, אָבִי שֶׁבַּשָּׁמַיִם, אָב הָרַחֲמָן, אָדוֹן כָּל הָעוֹלָמִים.

מֶרְחָבִים, מֶרְחָבִים, מֶרְחֲבֵי אֵל אִוְּתָה נַפְשִׁי. אַל תַּסְגִּירוּנִי בְּשׁוּם כְּלוּב, לֹא גַּשְׁמִי וְלֹא רוּחָנִי. שָׁטָה הִיא נִשְׁמָתִי בְּרַחֲבֵי שָׁמַיִם, לֹא יְכִילוּהָ קִירוֹת הַלֵּב, וְלֹא קִירוֹת מַעֲשֶׂה, מוּסָר, הִגָּיוֹן וְנִימוּס. מִמַּעַל לְכָל אֵלֶּה שָׁטָה הִיא וָעָפָה, מִמַּעַל לְכָל אֲשֶׁר יִקָּרֵא בְּכָל שֵׁם, מִמָּרוֹם מִכָּל עֹנֶג, מִכָּל נֹעַם וָיֹפִי, מִמָּרוֹם לְכָל נִשְׂגָּב וְנֶאֱצָל. "חוֹלַת אַהֲבָה אָנִי".

מִי יוֹדֵעַ אֶת עֹמֶק צַעֲרִי, מִי יוּכַל לְשַׁעֲרוֹ. הִנְנִי כָּלוּא בִּמְצָרִים רַבִּים, בִּגְבוּלִים שׁוֹנִים, וְרוּחִי שׁוֹאֵף לְמֶרְחָבִים נִשָּׂאִים. צָמְאָה נַפְשִׁי לֶאֱלֹהִים. אוֹר הָאֲצִילוּת הוּא חַיֵּי רוּחִי. אֱמוּנַת אֱלֹהִים גְּדוֹלָה בְּלֹא שׁוּם מַעְצוֹר טִבְעִי, הֶגְיוֹנִי, נִימוּסִי, מוּסָרִי, הוּא מְשׂוֹשׂ חַיַּי. כָּל מַה שֶּׁהוּא מֻגְדָּר, הֲרֵי הוּא חֻלִּין לְגַבֵּי הַקֹּדֶשׁ הָעֶלְיוֹן, אֲשֶׁר אוֹתוֹ אָנֹכִי מְבַקֵּשׁ. חוֹלַת אַהֲבָה אָנִי. מַה קָּשֶׁה לִי הַלִּמּוּד, מַה קָּשָׁה עָלַי הַהִסְתַּגְּלוּת אֶל הַפְּרָטִים. הַיָּמִים אֲנִי אוֹהֵב, הַשִּׁיטוֹת הַשְּׁמֵימִיוֹת - לָהֶם אֲנִי מִתְגַּעְגֵּעַ. "גַּל עֵינַי וְאַבִּיטָה נִפְלָאוֹת מִתּוֹרָתֶךָ".

לָמָּה זֶה אֲצַעֵר אֶת נִשְׁמָתִי שֶׁהִיא חָפֵצָה בַּחֹפֶשׁ - אִם הִיא מַרְגֶּשֶׁת אֶת עַצְמָהּ בַּת-חוֹרִין. לָמָּה זֶה אָשִׂים עַל רַגְלֶיהָ כְּבָלִים שֶׁל אֲסִירֵי עֹנִי, תָּשׁוּט בַּמֶרְחָבִים, תְּצַיֵּר, תִּתְרוֹמֵם כְּפִי מִדַּת שְׁאִיפָתָהּ, תַּעֲלֶה מַעְלָה מַעְלָה. וְאִם הִנְנִי מַרְגִּישׁ כִּי עֲווֹנוֹתַי הֵם הֵם אֲשֶׁר יֶאֶסְרוּהָ בַּעֲבֹתִים, מִיָּד אָבִיא לָהּ אֶת עֹז הַתְּשׁוּבָה, הַתְּשׁוּבָה הָעֶלְיוֹנָה, הַתְּשׁוּבָה שֶׁהִיא קְרוֹבָה בְּפִי וּבִלְבָבִי לַעֲשׂוֹתָהּ. וְהִרְהוּר הַתְּשׁוּבָה בְּרָז-מַחֲשַׁבְתוֹ מִיָּד יָאִיר אֶת כָּל מַחֲשַׁכֶּיהָ, וְיִתֵּן לָהּ גְּבוּרַת יֶשַׁע.

לֹא יִסּוֹג אָחוֹר לִבִּי מִכָּל עִצָּבוֹן שֶׁבָּעוֹלָם, כִּי אֵדַע שֶׁיֵּשׁ בָּזֶה מִשּׁוּם הִתְגַּלּוּתוֹ שֶׁל הַגַּרְעִין שֶׁבַּנֶפֶשׁ בְּכָל דַּרְגוֹתֶיהָ, מִצִּדָּהּ הַפְּרָטִי אוֹ מִצַּד יִחוּשָׂהּ אֶל הַכְּלָל. וְיִהְיוּ לִי כָּל הָעִצְּבוֹנוֹת לְהוֹרָאַת מְקוֹמוֹת הַמַּכְאוֹב, כְּדֵי לְתַקֵּן וּלְמַלֵּא אוֹר בַּמַּעֲשֶׂה, בַּשֵּׂכֶל, בָּרַעְיוֹן אוֹ בַּצִּיּוּר. וְכָל הַדְּרָכִים מְפַלְּשִׁים הֵם, כִּי חֶסֶד ד' גָּדוֹל מְאֹד. "כִּי-חַסְדְּךָ גָּדוֹל עָלָי וְהִצַּלְתָּ נַפְשִׁי מִשְּׁאוֹל תַּחְתִּיָּה".

הִתְעוֹרְרִי נַפְשִׁי, הִתְעוֹרְרִי בַּגְּדֻלָּה הָאֱלֹהִית שֶׁלָּךְ. אַל תִּתְבַּטְּלִי, וְלֹא תִּכָּנְעִי מִפְּנֵי גֵּאִים סוֹרְרִים, שֶׁאֵינָם יְכוֹלִים לְהַשִּׂיג אֶת הֵלֶךְ רוּחֵךְ וְעֶדְנוּת נַפְשֵׁךְ בְּצִיּוּרֶיהָ הַטְּהוֹרִים, וּבַעֲרִיגָתָהּ הָעֶלְיוֹנָה אֶל הַשִּׂגּוּב הֶעָדִין - הַשִּׂגּוּב הָאֱלֹהִי, וְאֶל הַיֹֹּשֶׁר, הָאֱמֶת וְהַטֹּהַר הַנַּעֲרָץ. דְּעִי אֶת עֶרְכֵּךְ וְהִתְרוֹמְמִי.
