מִמַּעְיָן קֹדֶשׂ-קֳדָשִׂים

נְחָלִים גְּדוֹלִים מִשְׁתַּפְּכִים בִּלְבָבִי, מַעְיָנוֹת גְּדוֹלִים רַחֲבֵי יָדָיִם נִפְתָּחִים לְפָנַי. הַתּוֹרָה, הַמִּצְווֹת וְאוֹרוֹתֵיהֶם, הוֹלְכִים וּמַזְרִיחִים עָלַי אֶת זֹהַר קַוֵּיהֶם. אַשְׁרֵי חֶלְקִי, כִּי אֱמוּנָה גְּדוֹלָה, אֱמוּנַת אֵל חֵי-הָעוֹלָמִים, אֵל אֱלֹהֵי יִשְׂרָאֵל, אֱלֹהֵי עוֹלָם ד', הִיא חֶלְקִי וְחֶבֶל נַחֲלָתִי, גּוֹרָלִי וְכוֹסִי, יִשְׁעִי וְחֶפְצִי. הַכֹּל מִצְטַיֵּר לְפָנַי בְּשִׁיטָה שְׁלֵמָה וּמַבְהֶקֶת. שְׂפָתִי קְצָרָה וַעֲיֵפָה מִגָּלוֹת עַל-יָדָהּ גַּם קְצֵה-קְצָתָהּ מֵרַעֲנַנּוּת אוֹר הַחַיִּים הָרוּחָנִיִּים, הַמְמַלְּאִים אֶת כָּל קְרָבַי. "כָּל עַצְמֹתַי תֹּאמַרְנָה ד' מִי כָמוֹךָ". "וַאֲנִי אָשִׁיר עֻזֶּךָ וַאֲרַנֵּן לַבֹּקֶר חַסְדֶּךָ".

מַזְהִירָה הִיא נִשְׁמַת הָעוֹלָם, מְלֵאָה הוֹד וְיִפְעָה הִיא, מְלֵאָה חַיִּים, גִּנְזֵי נְשָׁמוֹת, אוֹצְרוֹת רוּחַ הַקֹּדֶשׁ, מַעְיָנֵי גְּבוּרָה, גְּדֻלָּה וְתִפְאֶרֶת. הִנְנִי מִתְנַשֵּׂא בְּגָאוֹן לִמְרוֹם נִשְׁמַת הָעוֹלָמִים, חֵי-הָעוֹלָמִים. מַה נֶּהֱדָר הַמַּחֲזֶה, בֹּאוּ הִתְעַנְּגוּ, בֹּאוּ רְווּ נַחַת, הִתְעַדְּנוּ בַּעֲדָנִים. "טַעֲמוּ וּרְאוּ כִּי-טוֹב ד'". "לָמָּה תִשְׁקְלוּ-כֶסֶף בְּלוֹא-לֶחֶם וִיגִיעֲכֶם בְּלוֹא לְשָׂבְעָה, שִׁמְעוּ שָׁמוֹעַ אֵלַי וְאִכְלוּ-טוֹב וְתִתְעַנַּג בַּדֶּשֶׁן נַפְשְׁכֶם".

אֵינֶנִּי יָכוֹל לְחַלֵּק חִלּוּק מֻחְלָט בֵּין מָצוּי לְמָצוּי, כִּי-אִם חִלּוּק הַדְרָגִי. מִתְגַּלָּה לִי הַהֲוָיָה בְּכָל הֶקֵּפָהּ, מִיְצוּר הַיּוֹתֵר קָטָן עַד הַהוֹפָעָה הָרוּחָנִית הַמְּלֵאָה גְּבוּרָה, גְּדֻלָּה וְתִפְאֶרֶת, מְעֻטֶּרֶת בְּכֶתֶר חָכְמָה בִּינָה וָדַעַת, מַבְלֶטֶת בִּגְדֻלָּה אֶת נֶצַח-הַנְּצָחִים. וּבִגְבוּרָתָהּ הַמְאַגֶּדֶת אֶת כָּל הַהֲפָכִים, אֶת הַהוֹד הַהֲוָיָתִי, וּבְתִפְאַרְתָּהּ הַמָּלֵא דַּעַת, מְיַסֶּדֶת אֶת יְסוֹד הַכֹּל, מַפְרַחַת אֶת פְּרִיחַת הַכֹּל, כָּל הַנְּשָׁמוֹת, הָרוּחוֹת וְהַנְּפָשׁוֹת, כָּל הַתְּנוּעוֹת וְהַחֲפָצִים, הָרְצוֹנוֹת וְהַהַסְכָּמוֹת, כָּל הַשְּׁאִיפוֹת וְהַהִתְפַּעֲלֻיּוֹת, וּמַמְצִיאָתָם בְּכָל עֶרְכֵיהֶם בְּקֹבֶץ אֶחָד סִדּוּר מַמְלַכְתִּי. "ד' מֶלֶךְ".

... וּמְיַחֲדִים יִחוּדִים קְדוֹשִׁים, יִחוּדִים עֶלְיוֹנִים, בְּרִנָּה, בִּקְדֻשָּׁה, בְּשִׁירָה, בְּחֶדְוָה, בְּטָהֳרָה, בְּחִילָה, בִּרְעָדָה, בְּדִיצָה. וְכָל עַצְמוֹתַי תְּרַנֵּנָה, וּמַלְכוּת שָׁמַיִם מִתְעַלָּה, וְהַכָּבוֹד הָעֶלְיוֹן מִתְרוֹמֵם, וְנַפְשִׁי דּוֹרֶכֶת עֹז, וְהַלֵּל וְזִמְרָה הִנְנִי מִתְמַלֵּא, וְשָׁמַיִם וָאָרֶץ שְׂמֵחִים בְּשִׂמְחָתִי, וְכָל הַדִּינִים מִתְמַתְּקִים, וּשְׂרָפִים וְאַרְאֵלִים מְזַמְּרִים אִתִּי יַחַד, וּמִתְקַבְּצִים מַלְאֲכֵי הַשָּׁרֵת כִּבְמִזְמוּטֵי חָתָן וְכַלָּה, וְהַכֹּל אוֹמֵר כָּבוֹד. בָּרוּךְ כְּבוֹד אֲדֹנָי מִמְּקוֹמוֹ. בָּרוּךְ הוּא וּמְבֹרָךְ שְׁמוֹ הַגָּדוֹל, הַנֶּעֱלָם מִכָּל חַי וְחָבוּי מִכָּל יֵשׁ, וְעַל כֹּל יִתְגַּדֵּל וְיִתְנַשֵּׂא.

מָלֵא שִׂמְחָה אֲנִי, מָלֵא גְּדֻלָּה, מָלֵא שִׁפְלוּת, מָלֵא מְרִירוּת, מָלֵא נֹעַם, מָלֵא עֹנֶג, מָלֵא אַהֲבָה, מָלֵא קִנְאָה, מָלֵא זַעַם, מָלֵא חֶסֶד, מָלֵא טוֹב לַכֹּל. אַשְׁרֵי אָדָם שׁוֹמֵעַ לִי. אַשְׁרֵי נוֹתֵן לִי אֶת הָעֶרֶךְ הַפְּנִימִי הָרָאוּי לִי לְפִי חֶמְדַת סְגֻלָּתִי. הוּא יְרוֹמַם אַף יִנָּשֵׂא, יְשֻׂגָּב מִכָּל עֹנִי, יִתְקַדֵּשׁ וְיִטַּהֵר, וַד' אֱלֹהִים עִמּוֹ.

נָשִׁיר, נְזַמֵּר, נַרְגִּישׁ, נַשִּׂיג, נִבְכֶּה, נִרְקֹד, נַעֲלֹז. וַאֲנִי בַּד' אֶעְלוֹזָה אָגִילָה בֵּאלֹהֵי יִשְׁעִי, הַנּוֹתֵן נְקָמוֹת לִי וְעַל בָּמוֹתַי יַעֲמִידֵנִי. "מִגְדּוֹל יְשׁוּעוֹת מַלְכּוֹ וְעֹשֶׂה חֶסֶד לִמְשִׁיחוֹ לְדָוִד וּלְזַרְעוֹ עַד עוֹלָם".

אֵין לִי לְהִתְבַּיֵּשׁ וְלֹא לְהִתְפַּחֵד מִפְּנֵי חֵשֶׁק הַדְּבֵקוּת שֶׁלִּי, שֶׁהוּא, בָּרוּךְ ד', לִי טֶבַע קָבוּעַ מִשֹּׁרֶשׁ הַקְּדֻשָּׁה שֶׁבַּנְּשָׁמָה. וּלְהֶפֶךְ, כָּל מַה שֶּׁאֲנִי רוֹצֶה לִגְנֹז קְדֻשָּׁה זוֹ, הִנְנִי תּוֹעֶה וְנוֹפֵל. וְהִנְנִי חוֹזֵר וְנוֹשָׁע עַל-יְדֵי הִתְעוֹדְדוּת הַגַּדְלוּת הַפְּנִימִית שֶׁלִּי, לָדַעַת שֶׁאֲנִי צָרִיךְ לְשׁוֹטֵט לְמַעְלָה מִכָּל הַפְּרָטִים, וּלְהִדַּבֵּק תָּמִיד בְּשֹׁרֶשׁ הַקֹּדֶשׁ שֶׁל הַתּוֹרָה וְשֶׁל כָּל הָעוֹלָמוֹת. וְאֵין לִי לָחוּשׁ מִשּׁוּם לַעַג, וַד' יִהְיֶה בְּכִסְלִי וְשָׁמַר רַגְלִי מִלָּכֵד.

אֲנִי צָרִיךְ לְהִתְנַהֵג בִּנְמִיכוּת רוּחַ וּבְשִׁפְלוּת גְּדוֹלָה, וּבְכָל זֹאת חָלִילָה לְהִתְיָאֵשׁ מִכָּל גְּדֻלָּה, מִכָּל מַעֲלָה, מִכָּל בְּרָכָה, מִכָּל קְדֻשָּׁה, מִכָּל אוֹר תּוֹרָה, מִכָּל נֹעַם ד', מִכָּל הַצְלָחָה, מִכָּל תְּשׁוּעָה, מִכָּל שֶׁפַע קֹדֶשׁ, מִכָּל אוֹרָה וְצָהֳלָה, מִכָּל שִׂמְחָה וֶאֱמֶת. "אוֹר זָרֻעַ לַצַּדִּיק וּלְיִשְׁרֵי-לֵב שִׂמְחָה". "וְהִתְעַנַּג עַל-ד', וְיִתֶּן-לְךָ מִשְׁאֲלוֹת לִבֶּךָ. גּוֹל עַל-ד' דַּרְכֶּךָ וּבְטַח עָלָיו".

אֲנִי צָרִיךְ לְהַמְשִׁיךְ אֶת הָעֹנֶג הַקָּדוֹשׁ שֶׁל הַדְּבֵקוּת הָאֱלֹהִית הַשִּׂכְלִית בְּכָל מַה שֶּׁאֲנִי לוֹמֵד, עוֹשֶׂה, מִתְפַּלֵּל וּמְדַבֵּר. הַנִּגְלוֹת הַהִלְכָתִיּוֹת יֵעָשׂוּ רְטֻבִּים, מְלֵאֵי חַיֵּי נֹעַם שֶׁל קֹדֶשׁ עֶלְיוֹן. יְדִיעָה צְלוּלָה תַּחְדֹּר לְהָבִין לְהַשְׂכִּיל וּלְהַרְגִּישׁ, אֵיךְ כָּל הָאֳרָחוֹת הַסְּבוּכוֹת שֶׁל עִמְקֵי הַהֲלָכוֹת, הוֹלְכוֹת וּבָאוֹת מֵאוֹתוֹ הַמַּעְיָן הַמָּתוֹק וְהַנָּעִים, שֶׁכָּל הַתַּעֲנוּגִים הַקְּדוֹשִׁים וְכָל מִתְקֵי זִיוֵי הָעֲדָנִים נוֹבְעִים מִשָּׁם.

אֵינִי יָכוֹל לָזוּז מֵהַדְּבֵקוּת הָאֱלֹהִית, וְעַל-כֵּן אֲנִי מְחֻיָּב לְהִשְׁתַּדֵּל שֶׁאֶרְאֶה אֶת הָאוֹר הָאֱלֹהִי וְנָעֳמוֹ בְּכָל מִלֵּי-דִרְשׁוּת, בְּכָל דִּבּוּר וּבְכָל מַעֲשֶׂה וּתְנוּעָה, בֵּין שֶׁל עַצְמִי בֵּין שֶׁל אֲחֵרִים. וְקַל-וָחֹמֶר לְהַרְגִּישׁ אֶת גִלּוּי הָאוֹר הָעֶלְיוֹן דֶּרֶךְ צִנּוֹרוֹת הָאֱמֶת וְהַצֶּדֶק, בְּכָל הַתּוֹרָה כֻּלָּהּ, גַּם בִּפְשָׁטֶיהָ וּבַהֲוָיוֹת הַהִלְכָתִיּוֹת וְהַפִּלְפּוּלִים. וְזֶהוּ הַטַּעַם מַה שֶּׁאֲנִי נוֹטֶה תָּמִיד לְצַדֵּד אֶת הַפִּלְפּוּל בְּאֹפֶן הַהִגָּיוֹן הָרַעְיוֹנִי, שֶׁיֵּשׁ בּוֹ גַּם-כֵּן רֶגֶשׁ הַלֵּב, וְהַכֹּל הוּא מִתּוֹךְ הַנְּטִיָּה הַפְּנִימִית, שֶׁאֲנִי חָפֵץ בְּכָל לֵב, שֶׁהָאוֹר הָאֱלֹהִי בְּתַעֲנוּגוֹ וְאוֹרוֹ יִהְיֶה מְגֻלֶּה לַכֹּל, וְשֶׁהַכֹּל יִתְעַדְּנוּ וְיִתְעַנְּגוּ בּוֹ. וְהִנְנִי צָרִיךְ תָּמִיד אִמּוּץ תְּמִידִי שֶׁלֹּא לָזוּז מִמַּעֲלָה זוֹ, וּלְהַרְבּוֹת אֶת אוֹרָהּ וְלָסֹל יָפֶה אֶת מְסִלָּתָהּ, וּלְהַכְלִיל אֶת כָּל דַּרְכֵי הַלִּמּוּד וְהַהַנְהָגָה בְּדֶרֶךְ כְּלָלִי זֶה, שֶׁהוּא אֹרַח צַדִּיקִים הַהוֹלֵךְ וָאוֹר עַד נָכוֹן הַיּוֹם.

אֲנִי צָרִיךְ לְהַאֲמִין הַרְבֵּה מְאֹד בַּצַּד הַטּוֹב שֶׁלִּי, בִּנְקֻדַּת הַקֹּדֶשׁ שֶׁלִּי, בְּמַעְיָן הַחָכְמָה שֶׁהַקָּדוֹשׁ-בָּרוּךְ-הוּא מַשְׁפִּיעַ עָלַי עַל-יְדֵי קְדֻשַּׁת הָאוֹר מֵהַתּוֹרָה שֶׁשַּׁיָּךְ לִי, בְּכֹחַ הָרֶגֶשׁ הֶעָמֹק שֶׁל קְדֻשָּׁה וְשֶׁל אַהֲבַת הַבְּרִיּוֹת וְאַהֲבַת יִשְׂרָאֵל שֶׁלִּי, בְּקִשּׁוּרִי לְאַהֲבַת הָעוֹלָם-הַבָּא, לַחַיִּים הָרוּחָנִיִּים בִּפְנִים-פְּנִימִיּוּתִי. וְכָל אֱמוּנַת אֱמֶת אֹמֶן זֹאת, לֹא תַּשְׁכִּיחַ מִמֶּנִּי מְאוּמָה מִשִּׁפְלוּתִי וְדַלּוּתִי, מִקַּלּוּתִי וּבַעֲרוּתִי, מֵאֲפִיסַת עֶרְכִּי, מִשִּׁבְרוֹן לְבָבִי, מִמְּאִיסָתִי אֶת עַצְמִי, מֵחֶרְפָּתִי וּבָשְׁתִּי וּכְלִימָתִי. חֶרְפָּה שָׁבְרָה לִבִּי וָאֲנוּשָׁה, וָאֲקַוֶּה לָנוּד וָאַיִן, וְלִמְנַחֲמִים לֹא מָצָאתִי. בְּךָ ד' אֱלֹהִים עֵינַי, בְּךָ ד' חָסִיתִי אַל אֵבוֹשָׁה. אַל יֵבוֹשׁוּ בִּי קוֹוֶיךָ, ד' אֱלֹהִים צְבָאוֹת, אַל יִכָּלְמוּ בִּי מְבַקְּשֶׁיךָ. וּלְאִידָךְ גִּיסָא, פְּגָמַי הַגְּדוֹלִים, בְּזוּיֵי חֶרְפָּתִי, לֹא יִכְהוּ חַס-וְחָלִילָה אֲפִלּוּ בְּכָל-דְּהוּ מִקְּדֻשַּׁת הָאֱמוּנָה בַּחֵלֶק הַטּוֹב הָעֶלְיוֹן, מִמְּאוֹר הַקֹּדֶשׁ שֶׁבְּנִשְׁמַת אֱלֹהִים חַיִּים, שֶׁזִּקּוּקֵי אוֹרָן מְאִירִים בִּפְנִימִיּוּת רוּחִי וְנַפְשִׁי, וְלִבִּי וּבְשָׁרִי יְרַנְּנוּ לְאֵל חַי.

וְהָרָצוֹן הָאֵיתָן מִתְרוֹפֵף. צָרִיךְ אֲנִי לִשְׁאֹב מִמַּעְיָן הָעֶלְיוֹן שֶׁל קֹדֶשׁ-קֳדָשִׁים, שֶׁל גְּבוּרָה עֶלְיוֹנָה, עֹז יְשׁוּעוֹת, וְלַעֲמֹד תָּמִיד בְּדֵעָה קְבוּעָה בְּהֵיכָל הַקֹּדֶשׁ וּבְאוֹר הַחַיִּים הָאֲמִתִּיִּים, וּבְשִׂמְחַת ד' וְהַשָּׂגַת אוֹר עוֹלָם. וְכָל נִיצוֹץ מִתְלַקֵּט וּמִתּוֹסֵף לְאוֹר הָאֲבוּקָה הָעֶלְיוֹנָה, שֶׁמְּאִירָה אֶת דֶּרֶךְ הַחַיִּים לָנוּ וְלָעוֹלָם כֻּלּוֹ.

אַף-עַל-פִּי שֶׁנִּתְרַחַקְתִּי זְמָנִים רַבִּים מֵהוֹד הַקְּדֻשָּׁה הַפְּנִימִית, מִפְּנֵי הַרְגָּשַׁת הַחוֹבָה לִדְבָרִים מַעֲשִׂיִּים, וּבָהֶם גַּם לֹא נֶאֱחַזְתִּי כָּרָאוּי, בְּכָל יוֹם הִנְנִי מוּכָן לֶאֱחֹז בְּעֵץ-הַחַיִּים, וְלָשׁוּב לִשְׁקִיקַת הָאוֹר הָעֶלְיוֹן בְּנֹעַם ד' וְיִפְעַת חָכְמַת קֹדֶשׁ, לַמְרוֹת כָּל הַמַּפְרִיעִים מִבַּיִת וּמִחוּץ.

מִמְנוּחַת הַנֶּפֶשׁ וּמְעֲבוֹדָה פְּנִימִית, בְּכָל צוּרָה שֶׁהִיא, לֹא יוּכַל שׁוּם דָּבָר לְהַפְרִיעַ. אֲפִלּוּ מַה שֶּׁנִּתְגָּרַשְׁתִּי מֵאֶרֶץ הַקֹּדֶשׁ וְהִנְנִי שָׁרוּי בְּאֶרֶץ הָעַמִּים, עַד אֲשֶׁר יְרַחֲמֵנִי ד' מִמְּקוֹר הָרַחֲמִים, בָּרוּךְ הוּא, לְהָשִׁיבֵנִי לְאֶרֶץ הַקֹּדֶשׁ בְּרֹב-חֲסָדָיו. בְּכָל מָקוֹם מֻקְטָר מֻגָּשׁ לִשְׁמוֹ מִנְחָה טְהוֹרָה, בְּחִזּוּק וְאִמּוּץ.
