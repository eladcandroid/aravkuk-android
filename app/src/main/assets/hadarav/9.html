לַעֲבֹד עֲבוֹדַת ה'

חָפֵץ אֲנִי בִּתְשׁוּקָה וַחֲפִיצָה גְּמוּרָה לַעֲבֹד עֲבוֹדַת ד', עֲבוֹדַת קֹדֶשׁ עֶלְיוֹנָה לְמַעְלָה לְמַעְלָה מִכָּל חֹק שֶׁל טֶבַע. אֲנִי רוֹצֶה לְהַכְנִיס בְּחֵשֶׁק הַטֶּבַע כֻּלּוֹ אֶת הָאוֹר הַחֶפְצִי שֶׁל תְּשׁוּקַת קֹדֶשׁ הַקֳּדָשִׁים שֶׁל הָאִידֵאָלִיּוּת הָאֱלֹהִית. אֲנִי מִתְיַחֵד עִם כָּל פֶּרֶק-שִׁירָה, עִם כָּל הַמְשׁוֹרְרִים אֲשֶׁר בְּכָל הַיֵּשׁ, הָעוֹלִים בַּשִּׁיר וּמִתְעַלִּים בַּשִּׁיר וְנוֹתְנִים הוֹד וְהָדָר לִפְאֵר חֵי-הָעוֹלָמִים, בָּרוּךְ הוּא וּבָרוּךְ שְׁמוֹ.

לֹא אוּכַל לְהַכְחִיד בְּקִרְבִּי אֶת הָרוּחַ הַשּׁוֹקֵק לֶאֱחֹז בְּהָעֲבוֹדָה הָרוֹמְמָה שֶׁל הַעֲלָאַת הַנִּיצוֹצוֹת הַקְּדוֹשִׁים, שֶׁל כָּל הַמַּגִּיעַ לְחוּג הַשִּׁמּוּשׁ שֶׁל כָּל הַצְּרָכִים הַגַּשְׁמִיִּים, וְכָל הַקִּנְיָנִים שֶׁהֵם בָּאִים בְּמַגָּעִי וּגְבוּלִי. וְאוֹתָהּ קוֹרַת-הָרוּחַ הַמַּגִּיעָה לִי, אֲפִלּוּ עַל-פִּי הַדִּמְיוֹן הַכֵּהֶה מֵהַצִּיּוּר הַקָּדוֹשׁ שֶׁל עֲבוֹדָה תְּמִימָה וּקְדוֹשָׁה זוֹ, הוּא מְמַלְּאֵנִי עֹז וְתַעֲצוּמוֹת, שֶׁיָּבוֹאוּ לִכְלָל תִּקּוּן עַל-יְדֵי הִתְרַחֲבוּת אוֹר הַתּוֹרָה וְהָעֲבוֹדָה, בְּכָל הַצְּדָדִים הַהוֹלְכִים וּמִסְתַּעֲפִים. "יַגְדִּיל תּוֹרָה וְיַאְדִּיר".

לִי חֲסֵרָה אַהֲבַת עוֹלָם-הַזֶּה, וְזֶה מְסַבֵּב לִי חִסָּרוֹן גָּדוֹל בְּרוּחָנִיּוּת, וְגַם לִמּוּד תּוֹרָה וְעֵסֶק בַּמִּצְווֹת שֶׁלֹּא לִשְׁמָהּ - שֶׁמִּתּוֹךְ שֶׁלֹּא לִשְׁמָהּ בָּא לִשְׁמָהּ. וַאֲנִי צָרִיךְ הַדְרָכָה בְּאֵיזֶה אֹפֶן לֶאֱחֹז בַּדְּרָכִים הַלָּלוּ, שֶׁהֵם מִצַּד עַצְמָם יְרִידוֹת, מִכָּל-מָקוֹם הֵם צֹרֶךְ עֲלִיָּה.

נָבוֹךְ אֲנִי מְאֹד בְּעִנְיְנֵי אֲכִילָתִי. מִלְחָמָה גְּדוֹלָה יֵשׁ לִי בָּזֶה. מְאֹד הָיִיתִי רוֹצֶה לְהִתְעַנּוֹת, וּלְהִתְעַנּוֹת כָּרָאוּי, וְעַל-כָּל-פָּנִים לֶאֱכֹל מְעַט מְאֹד. וּלְאִידָךְ גִּיסָא צָרִיךְ אֲנִי, בֵּין מִצַּד הַבְּרִיאוּת בֵּין מִצַּד הַנְּטִיּוֹת הַקְּדוֹשׁוֹת - שֶׁהֵן גְּנוּזוֹת בְּמַעֲמַקַּי וִיכוֹלוֹת לִהְיוֹת לְאוֹרִים גְּדוֹלִים, לִדְבָרִים שֶׁכָּל גְּדוֹלֵי עוֹלָם מִתְנַשְּׂאִים עַל-יָדָם לִמְרוֹמֵי גְּבָהִים - מִצַּד כָּל אֵלֶּה צָרִיךְ אֲנִי לֶאֱכֹל כָּרָאוּי, וְלִפְעָמִים גַּם הַרְבֵּה. וּמִתּוֹךְ קִלְקוּל הָעִכּוּל שֶׁלִּי קָשְׁתָה לִי הָאֲכִילָה, וּמִתּוֹךְ פִּזּוּר הַנֶּפֶשׁ שֶׁאֲנִי סוֹבֵל עַל-פִּי-רֹב, קָשְׁתָה לִי הַהַכְרָעָה בְּסֵדֶר הָאֲכִילָה, וְיִשּׁוּב הַדַּעַת הַגָּמוּר הַנִּדְרָשׁ לִקְדֻשָּׁה שֶׁל חָכְמָה עֶלְיוֹנָה, הַנִּדְרָשׁ לָאֲכִילָה שֶׁבִּקְדֻשַּׁת אֱמֶת, הַנְּאוֹתָה לְטָהֳרַת הַנְּשָׁמָה, לְהַעֲלָאַת הָרָצוֹן לְפָאֵר וּלְרוֹמֵם קֹדֶשׁ ד' בְּעוֹלָמוֹ. וְאִם קָשֶׁה הַדָּבָר לִי בְּאֶרֶץ-יִשְׂרָאֵל, כַּמָּה קָשֶׁה הַדָּבָר בְּחוּץ-לָאָרֶץ, וּמִכָּל-מָקוֹם מְקַוֶּה אֲנִי לְחֶסֶד עֶלְיוֹן, שֶׁהַכֹּל יִתְחַדֵּשׁ לְטוֹבָה, לְאוֹר גָּדוֹל, לְתִפְאֶרֶת עוֹלָמִים, לִבְרָכָה וְלִישׁוּעָה, לְאוֹר תּוֹרָה וְנֵר מִצְוָה, וּבְמַעְגְּלֵי צֶדֶק יַדְרִיכֵנוּ צוּר יִשְׁעֵנוּ. יְפָאֵר עֲנָוִים בִּישׁוּעָה, וְאֶת דַּכָּא וּשְׁפַל רוּחַ, וְאֶת עַם עָנִי תּוֹשִיעַ.

עִקָּר חֶסְרוֹנִי הוּא חֻלְשַׁת הָרָצוֹן, וְגַם חֻלְשַׁת הַגּוּף. וְלִפְעָמִים כְּשֶׁאֲנִי רוֹצֶה לַעֲסֹק בִּתְשׁוּבָה וּבְיִרְאָה, נֶחֱלָשׁ יוֹתֵר הָרָצוֹן, וַאֲנִי צָרִיךְ אַחַר-כָּךְ לְתַקֵּן הַרְבֵּה יוֹתֵר. עַל-כֵּן אֲנִי צָרִיךְ לְמַזֵּג בְּתוֹךְ הַתְּשׁוּבָה גְּבוּרָה פְּנִימִית מְלֵאָה חֶסֶד, לְהַגְבִּיר וּלְחַזֵּק אֶת הָרָצוֹן, וּלְהִשְׁתַּדֵּל בְּחִזּוּק הַכֹּחַ הַגּוּפָנִי. וְעֶזְרַת ד' תּוֹפִיעַ עָלַי שֶׁיִּהְיֶה הַכֹּל בְּסֵדֶר נָכוֹן.

לְהַגְבִּיר הָרָצוֹן אֲנִי צָרִיךְ, לְשַׁחְרְרוֹ מִכָּל שִׁעְבּוּד שֶׁל דִּמְיוֹן. אָז יִהְיֶה קָשׁוּר יַחַד לַתּוֹרָה וְלַתְּפִלָּה, לִדְבֵקוּת אֱלֹהִית שִׂכְלִית, וּלְהִתְעוֹרְרוּת נַפְשִׁית בְּעֹמֶק הַקֹּדֶשׁ. וּזְכוּתָא דְּאֶרֶץ-יִשְׂרָאֵל תַּעֲמֹד לִי גַּם בְּגָלוּתִי, לְהָשִׁיבֵנִי לְאֶרֶץ הַקֹּדֶשׁ בְּרַחֲמִים רַבִּים.

אָנוּ צְרִיכִים תָּמִיד לְהַתְחִיל מֵרֹאשׁ, מֵרֵאשִׁית הַמַּחֲשָׁבוֹת שֶׁל הַיַּלְדוּת, לְבָרְרָן וּלְזַכְּכָן, וּלְהוֹצִיא אֶת הַטּוֹב וְהָאוֹר שֶׁבָּהֶן בְּטֹהַר וָאֹמֶץ. אֵין לָנוּ לְזַלְזֵל בְּמַחֲשָׁבוֹת פְּשׁוּטוֹת, הֵן הֵן עוֹמְדוֹת לָנוּ לְהָאִיר אֶת דַּרְכֵּנוּ יוֹתֵר מִכָּל הַמַּחֲשָׁבוֹת שֶׁאָנוּ מְדַמִּים שֶׁהֵן רָמוֹת וְנִשְׂגָּבוֹת.

בְּיַלְדוּתִי הָיִיתִי מַרְגִּישׁ סִרְחוֹן שֶׁל בֵּית הַכִּסֵּא מִבֵּית הַתִּפְלוּת שֶׁל הַגּוֹיִים, אַף-עַל-פִּי שֶׁהָיוּ לִכְאוֹרָה נְקִיִּים מְאֹד, וְעָמְדוּ בְּתוֹךְ גַּן שֶׁל עֵצִים נְטוּעִים.

שׂוֹנֵא אֲנִי אֶת הַגַּאֲוָה תַּכְלִית הַשִּׂנְאָה. הִיא אֵינָה מִתְעוֹרֶרֶת בְּקִרְבִּי מִצַּד עֶצֶם מָהוּתִי וְטִבְעִי. כָּל צְדָדֵי הַשְּׁלִילָה שֶׁלִּי הִנָּם עוֹמְדִים לְפָנַי תָּמִיד בְּעֹמֶק גִּלּוּיָם. רַק הַמַּעֲמָדִים הַחִיצוֹנִיִּים, הַהִתְעָרְבוּת עִם הַבְּרִיּוֹת, הַהֶכְרֵחַ שֶׁל הִתְקַטְּנוּת הַנְּשָׁמָה בִּשְׁבִיל צָרְכֵי הַחַיִּים הַחַבְרוּתִיִּים שֶׁל הֶהָמוֹן, הֵם מְעִירִים אֶת הַגֹּעַל שֶׁל תְּכוּנַת הַגַּאֲוָה, שֶׁצְּרִיכִים לָקַחַת חֵלֶק מַשֶּׁהוּ מִמֶּנָּה לְצֹרֶךְ הַהַנְהָגָה - לְהַמְעִיט לִפְעָמִים עַל-יָדָהּ אֶת הָאוֹרָה הָעֶלְיוֹנָה, כְּשֶׁהִיא מִתְפַּלֶּשֶׁת יוֹתֵר מֵהַשּׁוּרָה שֶׁיּוּכַל הַגּוּף בְּחֻלְשָׁתוֹ לְקַבְּלָהּ. אֲבָל חֶרְפָּתָהּ בְּעֶצֶם, וְהַכִּעוּר וְהַשֶּׁקֶר שֶׁבָּהּ, מֻכָּר לִי בְּעֹצֶם בְּהִירוּתוֹ.

מִטִּבְעִי אֲנִי נִתְבַּע לַעֲשׂוֹת כָּל הַדְּבָרִים לְשֵׁם פָּעֳלָם וּלְדַבֵּר בָּם לִשְׁמָם. וְכָל מִינֵי הַפְּנִיּוֹת וּפִגְמֵי הַמַּחֲשָׁבוֹת וְהָרְצוֹנוֹת, הִנָּם אֶצְלִי רַק צְדָדִים חִיצוֹנִיִּים, שֶׁאִם אֶתְחַזֵּק בֶּאֱמוּנָה גְּדוֹלָה בִּסְגֻּלַּת אוֹר נִשְׁמָתִי, הַמְאֻחֶדֶת עִם עֲנָוָה טְהוֹרָה אָז אֲנַצֵּחַ אֶת הַכֹּל, וְאוֹר ד' יָאִיר עָלַי בִּבְהִירוּת וְהַרְחָבָה, וְאוּכַל לְדַבֵּר דִּבְרֵי אֱמֶת בְּלֹא שׁוּם פַּחַד וּבְלֹא שׁוּם מַשּׂוֹא פָּנִים בָּעוֹלָם.

אִם אֲנִי מֻכְרָח לִהְיוֹת אִישׁ רִיב לְכָל הָעוֹלָם, מִצַּד הַנְּטִיָּה שֶׁל הָאֱמֶת הָעֲמֻקָּה שֶׁבְּנַפְשִׁי, שֶׁאֵינוֹ סוֹבֵל שׁוּם הַטָּיָה שֶׁל שֶׁקֶר, אִי אֶפְשָׁר לִי לִהְיוֹת אִישׁ אַחֵר, וְצָרִיךְ אֲנִי לְהוֹצִיא מִן הַכֹּחַ אֶל הַפֹּעַל רַק אֶת יְסוֹדוֹת הָאֱמֶת הָעִקָּרִיִים הַצְּפוּנִים בְּרוּחִי.

הוֹשִׁיעֵנִי ד', פְּדֵנִי מֵעֹשֶׁק אָדָם, זַכֵּנִי לְצַחְצֵחַ כָּל מָהוּתִי בְּצַחְצְחוֹת אֱמֶת. חַזֵּק וְאַמֵּץ רְצוֹנִי וְכֹחוֹתַי לָלֶכֶת בַּמְּסִלָּה הַיְשָׁרָה שֶׁל אוֹר קְדֻשַּׁת יָשְׁרֶךָ, כִּי מֵישָׁרִים תִּרְצֶה, אַתָּה ד' אֱלֹהַי.

הַזִּכּוּכִים שֶׁאֲנִי מְזַכֵּךְ אֶת עַצְמִי, אֶת מַחְשְׁבוֹתַי, רַעְיוֹנוֹתַי, מִדּוֹתַי וְהַרְגָּשׁוֹתַי, הֵם יִהְיוּ זִכּוּכִים כּוֹלְלִים לְכָל הָעוֹלָם כֻּלּוֹ. "חַיָּב אָדָם לוֹמַר: בִּשְׁבִילִי נִבְרָא הָעוֹלָם".
